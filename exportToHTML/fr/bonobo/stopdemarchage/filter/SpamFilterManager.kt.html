package fr.bonobo.stopdemarchage.filter

import android.content.Context
import android.content.SharedPreferences
import com.google.gson.Gson
import com.google.gson.reflect.TypeToken
import fr.bonobo.stopdemarchage.utils.CountryDetector
import java.io.InputStreamReader
import fr.bonobo.stopdemarchage.filter.*

data class BlockedWordsConfig(
    val version: String,
    val country: String,
    val last_updated: String,
    val keywords: List<String>,
    val suspicious_patterns: List<SuspiciousPattern>? = null,
    val trusted_senders: List<String>? = null
)

data class SuspiciousPattern(
    val pattern: String,
    val description: String
)

data class FilterStats(
    val country: String,
    val keywordCount: Int,
    val patternCount: Int,
    val trustedSenderCount: Int,
    val lastUpdated: String,
    val autoDetectEnabled: Boolean
)

data class SpamCheckResult(
    val isSpam: Boolean,
    val reason: String,
    val confidence: Double,
    val blockedByPrefix: Boolean,
    val blockedByKeyword: Boolean,
    val blockedByPattern: Boolean = false,
    val matchedKeywords: List<String> = emptyList()
)

class SpamFilterManager(private val context: Context) {

    private val prefs: SharedPreferences =
        context.getSharedPreferences("spam_filter_prefs", Context.MODE_PRIVATE)

    private val gson = Gson()
    private val countryDetector = CountryDetector(context)

    companion object {
        private const val PREF_SELECTED_COUNTRY = "selected_country"
        private const val PREF_AUTO_DETECT = "auto_detect_country"
    }

    /**
     * Initialise le filtre au premier lancement
     */
    fun initialize() {
        if (!prefs.contains(PREF_SELECTED_COUNTRY)) {
            // Première installation: détection automatique
            val detectedCountry = countryDetector.detectCountry()
            setCountry(detectedCountry, autoDetect = true)
        }
    }

    /**
     * Obtient le pays actuellement configuré
     */
    fun getCurrentCountry(): String {
        return prefs.getString(PREF_SELECTED_COUNTRY, CountryDetector.DEFAULT_COUNTRY)
            ?: CountryDetector.DEFAULT_COUNTRY
    }

    /**
     * Définit le pays manuellement
     */
    fun setCountry(country: String, autoDetect: Boolean = false) {
        prefs.edit()
            .putString(PREF_SELECTED_COUNTRY, country)
            .putBoolean(PREF_AUTO_DETECT, autoDetect)
            .apply()
    }

    /**
     * Vérifie si la détection automatique est activée
     */
    fun isAutoDetectEnabled(): Boolean {
        return prefs.getBoolean(PREF_AUTO_DETECT, true)
    }

    /**
     * Charge la configuration des mots bloqués pour le pays actuel
     */
    fun loadBlockedWords(): BlockedWordsConfig {
        val country = getCurrentCountry()
        val fileName = when (country) {
            CountryDetector.COUNTRY_BELGIUM -> "blocked_words_be.json"
            CountryDetector.COUNTRY_FRANCE -> "blocked_words_fr.json"
            else -> "blocked_words_fr.json"
        }

        return loadConfigFromAssets(fileName)
    }

    /**
     * Charge un fichier de configuration depuis les assets
     */
    private fun loadConfigFromAssets(fileName: String): BlockedWordsConfig {
        return try {
            val inputStream = context.assets.open(fileName)
            val reader = InputStreamReader(inputStream)
            val type = object : TypeToken<BlockedWordsConfig>() {}.type
            gson.fromJson(reader, type)
        } catch (e: Exception) {
            // Fallback avec liste vide en cas d'erreur
            BlockedWordsConfig(
                version = "1.0",
                country = "UNKNOWN",
                last_updated = "",
                keywords = emptyList()
            )
        }
    }

    /**
     * Vérifie si un SMS est suspect (combinaison préfixes + mots-clés)
     */
    fun isSpam(message: String, sender: String? = null): SpamCheckResult {
        val config = loadBlockedWords()
        val messageLower = message.lowercase()
        val prefixFilter = PrefixFilterManager(context)

        // Vérifier d'abord le numéro expéditeur par préfixe
        var blockedByPrefix = false
        var prefixReason: String? = null

        sender?.let {
            val blockResult = prefixFilter.shouldBlockByPrefix(it, getCurrentCountry())
            if (blockResult.shouldBlock) {
                blockedByPrefix = true
                prefixReason = blockResult.reason
            }

            // Vérifier les expéditeurs de confiance
            if (config.trusted_senders?.any { trusted ->
                    it.equals(trusted, ignoreCase = true)
                } == true) {
                return SpamCheckResult(
                    isSpam = false,
                    reason = "Expéditeur de confiance",
                    confidence = 0.0,
                    blockedByPrefix = false,
                    blockedByKeyword = false
                )
            }
        }

        // Vérifier les mots-clés
        val blockedKeywords = mutableListOf<String>()
        config.keywords.forEach { keyword ->
            if (messageLower.contains(keyword.lowercase())) {
                blockedKeywords.add(keyword)
            }
        }

        val blockedByKeyword = blockedKeywords.isNotEmpty()

        // Vérifier les patterns regex suspects
        var blockedByPattern = false
        var patternMatch: String? = null

        config.suspicious_patterns?.forEach { pattern ->
            try {
                if (Regex(pattern.pattern).containsMatchIn(message)) {
                    blockedByPattern = true
                    patternMatch = pattern.description
                }
            } catch (e: Exception) {
                // Ignorer les regex invalides
            }
        }

        // Calculer le score de confiance
        var confidence = 0.0
        val reasons = mutableListOf<String>()

        if (blockedByPrefix) {
            confidence += 0.4
            reasons.add(prefixReason ?: "Préfixe suspect")
        }

        if (blockedByKeyword) {
            confidence += 0.3 * (blockedKeywords.size.coerceAtMost(3) / 3.0)
            reasons.add("${blockedKeywords.size} mot(s)-clé(s) suspect(s)")
        }

        if (blockedByPattern) {
            confidence += 0.3
            reasons.add(patternMatch ?: "Pattern suspect")
        }

        val isSpam = confidence >= 0.5

        return SpamCheckResult(
            isSpam = isSpam,
            reason = reasons.joinToString(", "),
            confidence = confidence,
            blockedByPrefix = blockedByPrefix,
            blockedByKeyword = blockedByKeyword,
            blockedByPattern = blockedByPattern,
            matchedKeywords = blockedKeywords
        )
    }

    /**
     * Met à jour automatiquement le pays si la détection auto est activée
     */
    fun updateCountryIfNeeded() {
        if (isAutoDetectEnabled()) {
            val detectedCountry = countryDetector.detectCountry()
            val currentCountry = getCurrentCountry()

            if (detectedCountry != currentCountry) {
                setCountry(detectedCountry, autoDetect = true)
            }
        }
    }

    /**
     * Obtient des statistiques sur les mots bloqués
     */
    fun getFilterStats(): FilterStats {
        val config = loadBlockedWords()
        return FilterStats(
            country = config.country ?: "",
            keywordCount = config.keywords?.size ?: 0,
            patternCount = config.suspicious_patterns?.size ?: 0,
            trustedSenderCount = config.trusted_senders?.size ?: 0,
            // CORRECTION FINALE : Remplace 0L par une chaîne vide "" ou une date par défaut String
            lastUpdated = config.last_updated ?: "",
            autoDetectEnabled = isAutoDetectEnabled()
        )
    }

}
