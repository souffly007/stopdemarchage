package fr.bonobo.stopdemarchage.utils

import android.content.Context
import android.telephony.TelephonyManager
import java.util.Locale

class CountryDetector(private val context: Context) {

    companion object {
        const val COUNTRY_FRANCE = "FR"
        const val COUNTRY_BELGIUM = "BE"
        const val DEFAULT_COUNTRY = COUNTRY_FRANCE
    }

    /**
     * Détecte automatiquement le pays de l'utilisateur
     * @return Code pays ISO (FR ou BE)
     */
    fun detectCountry(): String {
        // Priorité 1: Réseau mobile (plus fiable)
        val networkCountry = getNetworkCountry()
        if (networkCountry != null) {
            return networkCountry
        }

        // Priorité 2: SIM card
        val simCountry = getSimCountry()
        if (simCountry != null) {
            return simCountry
        }

        // Priorité 3: Locale système
        val localeCountry = getLocaleCountry()
        if (localeCountry != null) {
            return localeCountry
        }

        // Par défaut: France
        return DEFAULT_COUNTRY
    }

    /**
     * Détecte le pays via le réseau mobile
     */
    private fun getNetworkCountry(): String? {
        return try {
            val telephonyManager = context.getSystemService(Context.TELEPHONY_SERVICE) as TelephonyManager
            val countryCode = telephonyManager.networkCountryIso?.uppercase()

            when (countryCode) {
                "BE" -> COUNTRY_BELGIUM
                "FR" -> COUNTRY_FRANCE
                else -> null
            }
        } catch (e: Exception) {
            null
        }
    }

    /**
     * Détecte le pays via la carte SIM
     */
    private fun getSimCountry(): String? {
        return try {
            val telephonyManager = context.getSystemService(Context.TELEPHONY_SERVICE) as TelephonyManager
            val countryCode = telephonyManager.simCountryIso?.uppercase()

            when (countryCode) {
                "BE" -> COUNTRY_BELGIUM
                "FR" -> COUNTRY_FRANCE
                else -> null
            }
        } catch (e: Exception) {
            null
        }
    }

    /**
     * Détecte le pays via la locale système
     */
    private fun getLocaleCountry(): String? {
        val countryCode = Locale.getDefault().country.uppercase()

        return when (countryCode) {
            "BE" -> COUNTRY_BELGIUM
            "FR" -> COUNTRY_FRANCE
            else -> null
        }
    }

    /**
     * Vérifie si un numéro de téléphone est belge
     */
    fun isBelgianNumber(phoneNumber: String): Boolean {
        val cleanNumber = phoneNumber.replace(Regex("[^0-9+]"), "")

        return cleanNumber.startsWith("+32") ||
                cleanNumber.startsWith("0032") ||
                (cleanNumber.startsWith("0") && cleanNumber.length == 10)
    }

    /**
     * Vérifie si un numéro de téléphone est français
     */
    fun isFrenchNumber(phoneNumber: String): Boolean {
        val cleanNumber = phoneNumber.replace(Regex("[^0-9+]"), "")

        return cleanNumber.startsWith("+33") ||
                cleanNumber.startsWith("0033") ||
                (cleanNumber.startsWith("0") && cleanNumber.length == 10)
    }

    /**
     * Détermine le pays basé sur le numéro de téléphone
     */
    fun getCountryFromPhoneNumber(phoneNumber: String): String {
        return when {
            isBelgianNumber(phoneNumber) -> COUNTRY_BELGIUM
            isFrenchNumber(phoneNumber) -> COUNTRY_FRANCE
            else -> detectCountry()
        }
    }
}
